Custom Routes
get '/hello/:name' do
  "Hello #{params[:name]}!"
end



get '/say/*/to/*/download/*.*' do
  params[:splat] # => ["hello", "world, path/to/file", "xml"]
end


get '/posts.?:format?', :provides => ['rss', 'atom', 'xml'] do
  # matches any extension "GET /posts.json", "GET /posts.xml"..
